1.找出数组中重复的数字

 1function findRepeatNum(Arr) {
 2    let len = Arr.length;
 3    let result = [];
 4    let set = new Set();
 5    for(let i = 0; i < len; i++) {
 6        let setSize = set.size;
 7        set.add(Arr[i]);
 8        if(setSize == set.size) {
 9            result.push(Arr[i]);
10        }
11    }
12    return Array.from(new Set(result));
13}
14console.log(findRepeatNum([1,32,3,5,6,2,2,3,5,6,7]));  //[ 2, 3, 5, 6 ]

2.找出数组中第K大的数字(包括重复的元素，不用去重)

1function findKth(a, n, K) {
2    a.sort()
3    return a[n-K];
4}
5console.log(findKth([1,3,5,2,2],5,3));
